textSolution :: (Reflex t, MonadFix m, MonadHold t m)
             => Behavior t Int
             -> Event t Text
             -> m (Behavior t [Text], Event t [Text])
textSolution bIn eIn = mdo
  let
    add n xs x = take n (x : xs)
    e = add <$> bIn <*> b <@> eIn
  b <- _
  pure (b, e)
