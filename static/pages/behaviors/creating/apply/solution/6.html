applySolution :: (Reflex t, MonadHold t m)
              => Event t Int
              -> m (Behavior t Int, Behavior t Int)
applySolution eIn = do
  let
    multiple n =
      (== 0) . (`mod` n)
    downsample n =
      fmap (`div` n) . ffilter (multiple n)
  b2 <- hold 0 $ downsample 2 eIn
  b3 <- hold 0 $ downsample 3 eIn
  pure (b2 , b3)
