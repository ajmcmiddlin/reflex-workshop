applySolution :: (Reflex t, MonadHold t m)
              => Event t Int
              -> m (Behavior t Int, Behavior t Int)
applySolution eIn = do
  let
    e2 = (`div` 2) <$> eIn
    e3 = (`div` 3) <$> eIn
  h2 <- hold 0 e2
  h3 <- hold 0 e3
  pure (_ , _ )
