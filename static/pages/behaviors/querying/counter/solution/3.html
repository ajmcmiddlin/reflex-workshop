counterSolution' :: Reflex t
                 => Behavior t Int
                 -> Event t Int
                 -> Event t ()
                 -> Event t Int
counterSolution' bCount eAdd eReset =
  leftmost [
      0   <$             eReset
    , (+) <$> bCount <@> eAdd
    ]

If eAdd fires, we want to fire the result with the sum of the value of the Event and the value of bCount at the moment that eAdd fired.
