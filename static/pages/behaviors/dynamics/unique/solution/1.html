uniqueSolution :: (Reflex t, MonadFix m, MonadHold t m)
               => Dynamic t (Int, Int)
               -> m (Dynamic t Int, Dynamic t Int)
uniqueSolution dIn = do
  let
    dOut1 =              fst <$> dIn
    dOut2 =              snd <$> dIn
  pure (dOut1      , dOut2      )
