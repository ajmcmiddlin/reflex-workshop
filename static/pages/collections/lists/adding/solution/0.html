addItem :: MonadWidget t m
        => m (Event t TodoItem)
addItem =  do
  ti <- textInput $
    def

  let
    eEnter    = keypress Enter ti
    bText     = current $ value ti
    eStripped = Text.strip <$> bText <@ eEnter
    eText     = ffilter (not . Text.null) eStripped
    eAdd      = TodoItem False <$> eText

  pure eAdd
