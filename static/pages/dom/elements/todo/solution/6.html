todoItem :: MonadWidget t m
         => TodoItem
         -> m (Event t ())
todoItem item = 
  divClass "d-flex flex-row align-items-center" $ do
    let iText = item ^. todoItem_text
    divClass "p-1" $
      text iText
    divClass "p-1" $
      button "x"
