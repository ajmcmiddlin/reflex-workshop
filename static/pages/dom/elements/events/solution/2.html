eventsSolution :: MonadWidget t m
               => Dynamic t Bool
               -> m a
               -> m (Event t ())
eventsSolution dIn w = do
  let
    dAttrs = bool mempty ("hidden" =: "") <$> dIn
    attrs = "class" =: "text-uppercase"
  (e, _) <- elDynAttr' "div" (pure attrs <> dAttrs) $
    w
  pure never
