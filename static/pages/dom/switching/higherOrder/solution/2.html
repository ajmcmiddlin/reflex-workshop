higherOrderSolution :: (Reflex t, MonadFix m, MonadHold t m)
                    => Event t Bool
                    -> Event t ()
                    -> m (Dynamic t Int)
higherOrderSolution eClickable eClick = do
  let
    eReset = const 0 <$ eClick
    eAdd   = (+ 1)   <$ eClick

    eeMode = bool eReset eAdd <$> eClickable





  _
