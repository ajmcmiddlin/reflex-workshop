fmapMaybeSolution :: Reflex t 
                  => Event t Text 
                  -> (Event t Text, Event t Int)
fmapMaybeSolution eIn =
  let
    eOut   = -- Event t Int
      fmapMaybe convert eIn
    eError = -- Event t Text
      _            <$ difference  _          _
  in
    (eError, eOut)

We're going to use difference to report on errors.
