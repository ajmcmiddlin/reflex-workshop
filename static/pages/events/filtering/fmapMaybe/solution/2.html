fmapMaybeSolution :: Reflex t 
                  => Event t Text 
                  -> (Event t Text, Event t Int)
fmapMaybeSolution eIn =
  let
    eMaybe =  -- Event t (Maybe Int)
      convert <$> eIn
    eOut   =  -- Event t Int
      fmapMaybe id eMaybe


  in
    (_     , eOut)

We then use fmapMaybe id to filter out and remove the Nothing values for the success case...
