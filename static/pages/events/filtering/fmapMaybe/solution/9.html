fmapMaybeSolution :: Reflex t 
                  => Event t Text 
                  -> (Event t Text, Event t Int)
fmapMaybeSolution eIn =
  let
    eOut   = -- Event t Int
      fmapMaybe convert eIn
    eError = -- Event t Text
      "Not an Int" <$ difference (void eIn) (void eOut)
  in
    (eError, eOut)
