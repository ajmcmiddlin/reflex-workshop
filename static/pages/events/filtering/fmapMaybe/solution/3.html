fmapMaybeSolution :: Reflex t 
                  => Event t Text 
                  -> (Event t Text, Event t Int)
fmapMaybeSolution eIn =
  let
    eMaybe =  -- Event t (Maybe Int)
      convert <$> eIn
    eOut   =  -- Event t Int
      fmapMaybe id eMaybe
    eError =  -- Event t Text
      "Not an Int" <$ ffilter isNothing eMaybe
  in
    (eError, eOut)

... and ffilter isNothing to deal with the failure case
