leftmostSolution :: Reflex t
                 => Event t Int
                 -> ( Event t Text
                    , Event t Text
                    , Event t Text
                    , Event t Text
                    )
leftmostSolution eIn =
  let
    multiple n = (== 0) . (`mod` n)
    eFizz = "Fizz" <$ ffilter (multiple 3) eIn
    eBuzz = "Buzz" <$ ffilter (multiple 5) eIn
    eFizzBuzz = eFizz <> eBuzz
    tshow = Text.pack . show

  in
    (eFizz, eBuzz, eFizzBuzz, _        )
